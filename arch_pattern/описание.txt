install
    pip install gunicorn


Запуск
    gunicorn -b 127.0.0.1:8080 server:process_http_request
        -bили --bind означает “привязка” к определённому IP-адресу и порту.
        0.0.0.0  на всех сетевых интерфейсах
        server — название вашего скрипта без .py
        process_http_request — название функции


Демон gunicorn
    cd /etc/systemd/system 
    nano getip.service
        [Unit]
        Description=GetIP site

        [Service]
        Type=simple
        WorkingDirectory=/home/gas53/Documents/arch_pattern/  # путь к каталогу
        ExecStart=gunicorn -b 127.0.0.1:8080 server:process_http_request
        Restart=always

        [Install]
        WantedBy=multi-user.target


    systemctl daemon-reload
    systemctl start getip.service
    systemctl status getip
